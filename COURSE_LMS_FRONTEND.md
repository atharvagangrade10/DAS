# Course/LMS Frontend Development Document

## Overview
This document describes the frontend implementation for the Learning Management System (LMS) where Participants progress through Courses with multiple Steps. Each Step has learning Materials followed by an Exam. Participants must pass the step exam (approved by Manager) to unlock the next step.

## Key Features
- **Courses**: Multiple courses linked in sequence (like stairs)
- **Steps**: Each course has multiple steps/levels
- **Materials**: Videos, Books, Documents, and **Requirement Tables** (interactive checklists)
- **Step Exams**: Each step has its own exam (MCQ + Theory) that must be passed to progress
- **Manager Review**: Managers approve requirement tables and grade step exams

## Tech Stack Recommendations

- **Framework**: React.js / Next.js
- **State Management**: Zustand / Redux Toolkit / Context API
- **Forms**: React Hook Form
- **HTTP Client**: Axios / Fetch API
- **UI Components**:
  - Material UI (MUI) / shadcn/ui / Ant Design
  - Tailwind CSS for styling
- **Video Player**: React Player
- **Rich Text Editor**: Tiptap / Lexical (for theory answers)

## API Service

```typescript
// services/api.ts
import axios from 'axios';

const api = axios.create({
  baseURL: process.env.API_URL || 'http://localhost:8000',
  headers: {
    'Content-Type': 'application/json',
  },
});

// Add auth interceptor
api.interceptors.request.use((config) => {
  const token = localStorage.getItem('token');
  if (token) {
    config.headers.Authorization = `Bearer ${token}`;
  }
  return config;
});

// ============================================
// COURSE APIs
// ============================================

export const courseApi = {
  // Get all courses
  getAll: () => api.get('/courses'),

  // Get course by ID with steps
  getById: (id: string) => api.get(`/courses/${id}`),

  // Create new course
  create: (data: {
    name: string;
    description?: string;
    image_url?: string;
    is_default?: boolean;
    previous_course_id?: string;
    order_index: number;
  }) => api.post('/courses', data),

  // Update course
  update: (id: string, data: {
    name?: string;
    description?: string;
    image_url?: string;
    is_default?: boolean;
    previous_course_id?: string;
    next_course_id?: string;
    is_active?: boolean;
  }) => api.put(`/courses/${id}`, data),

  // Delete course
  delete: (id: string) => api.delete(`/courses/${id}`),

  // Set as default course
  setDefault: (id: string) => api.put(`/courses/${id}/set-default`),
};

// ============================================
// STEP APIs
// ============================================

export const stepApi = {
  // Create step in course
  create: (courseId: string, data: {
    name: string;
    description?: string;
    order_index: number;
    previous_step_id?: string;
    unlock_on_exam_pass?: boolean;
  }) => api.post(`/courses/${courseId}/steps`, data),

  // Update step
  update: (stepId: string, data: {
    name?: string;
    description?: string;
    order_index?: number;
    exam_id?: string;
  }) => api.put(`/steps/${stepId}`, data),

  // Delete step
  delete: (stepId: string) => api.delete(`/steps/${stepId}`),

  // Reorder steps
  reorder: (courseId: string, orders: Array<{
    step_id: string;
    order_index: number;
  }>) => api.put(`/courses/${courseId}/steps/reorder`, { step_orders: orders }),

  // Link exam to step
  linkExam: (stepId: string, formId: string) =>
    api.put(`/steps/${stepId}/exam`, { form_id: formId }),

  // Get step details (for participant)
  getDetails: (stepId: string) => api.get(`/steps/${stepId}`),

  // Mark materials as complete (unlock exam)
  markMaterialsComplete: (stepId: string) =>
    api.post(`/steps/${stepId}/materials-complete`),

  // Start step exam
  startExam: (stepId: string) => api.post(`/steps/${stepId}/exam/start`),
};

// MATERIAL APIs

export const materialApi = {
  // Create material in step
  create: (stepId: string, data: {
    material_type: 'RequirementTable' | 'Video' | 'BookLink' | 'Document' | 'Audio';
    title: string;
    description?: string;
    content: Record<string, any>;
    order_index: number;
    is_mandatory?: boolean;
  }) => api.post(`/steps/${stepId}/materials`, data),

  // Update material
  update: (materialId: string, data: {
    title?: string;
    description?: string;
    content?: Record<string, any>;
  }) => api.put(`/materials/${materialId}`, data),

  // Delete material
  delete: (materialId: string) => api.delete(`/materials/${materialId}`),

  // Mark material as complete (for videos, books, etc.)
  markComplete: (materialId: string) =>
    api.post(`/materials/${materialId}/complete`),
};

export const requirementTableApi = {
  // Manager: Create requirement table
  create: (data: {
    name: string;
    description?: string;
    step_id?: string;
    requirements: Array<{
      serial_number: number;
      activity: string;
      target: number;
      unit?: string;
    }>;
  }) => api.post('/requirement-tables', data),

  // Manager: Update requirement table
  update: (tableId: string, data: {
    name?: string;
    requirements?: Array<{
      serial_number: number;
      activity: string;
      target: number;
      unit?: string;
    }>;
  }) => api.put(`/requirement-tables/${tableId}`, data),

  // Get requirement tables
  getAll: (stepId?: string) =>
    api.get('/requirement-tables', { params: { step_id: stepId } }),

  // Participant: Get my submission
  getMySubmission: (tableId: string) =>
    api.get(`/requirement-tables/${tableId}/my-submission`),

  // Participant: Update row status
  updateRow: (submissionId: string, data: {
    serial_number: number;
    completed: boolean;
    completed_value?: number;
    remarks?: string;
  }) => api.post(`/requirement-table-submissions/${submissionId}/rows`, data),

  // Participant: Submit for approval
  submitForApproval: (submissionId: string) =>
    api.post(`/requirement-table-submissions/${submissionId}/submit`),

  // Manager: Get pending submissions
  getPending: (stepId?: string) =>
    api.get('/requirement-table-submissions/pending', { params: { step_id: stepId } }),

  // Manager: Approve/reject row
  reviewRow: (submissionId: string, serialNumber: number, data: {
    approved: boolean;
    approval_remarks?: string;
  }) => api.post(`/requirement-table-submissions/${submissionId}/rows/${serialNumber}/approve`, data),

  // Manager: Approve entire table
  approveTable: (submissionId: string) =>
    api.post(`/requirement-table-submissions/${submissionId}/approve`),
};

export const stepExamApi = {
  // Manager: Create step exam
  create: (data: {
    name: string;
    description?: string;
    step_id: string;
    duration_minutes: number;
    passing_marks: number;
    total_marks: number;
    max_attempts: number;
    passing_percentage: number;
    questions: Array<{
      question_id?: string;
      question_type: 'MCQ' | 'Theory';
      question_text: string;
      options?: Array<{ option_id: string; text: string; is_correct: boolean }>;
      marks: number;
      max_words?: number;
    }>;
  }) => api.post('/forms', data),

  // Get exam by ID
  getById: (formId: string) => api.get(`/forms/${formId}`),

  // Start exam (returns submission with questions)
  start: (stepId: string) => api.post(`/steps/${stepId}/exam/start`),

  // Submit exam answers
  submit: (submissionId: string, data: {
    mcq_answers: Array<{ question_id: string; selected_option_id: string }>;
    theory_answers: Array<{ question_id: string; answer_text: string }>;
  }) => api.post(`/form-submissions/${submissionId}/submit`, data),

  // Get my exam results
  getMyResult: (submissionId: string) => api.get(`/form-submissions/${submissionId}`),
};

export const reviewApi = {
  // Get pending exam submissions
  getPendingExams: (stepId?: string, courseId?: string) =>
    api.get('/form-submissions/pending', {
      params: { step_id: stepId, course_id: courseId }
    }),

  // Get submission details for grading
  getSubmission: (id: string) => api.get(`/form-submissions/${id}`),

  // Grade theory answers
  gradeTheory: (id: string, data: {
    theory_grades: Array<{ question_id: string; marks_awarded: number }>;
    feedback?: string;
  }) => api.post(`/form-submissions/${id}/grade`, data),

  // Approve submission (unlocks next step)
  approve: (id: string, data: {
    feedback?: string;
    unlock_next_course?: boolean; // For final step
  }) => api.post(`/form-submissions/${id}/approve`, data),

  // Reject submission
  reject: (id: string, data: { feedback?: string }) =>
    api.post(`/form-submissions/${id}/reject`, data),

  // Request retake
  requestRetake: (id: string, data: { reason: string }) =>
    api.post(`/form-submissions/${id}/request-retake`, data),
};
```
